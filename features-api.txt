Endpoints

    /api
        /register
            {name, username, password, email, contact, work}
                {token}
                {error: 'username'} (Username already taken)

        /login
            {username, password}
                {token}
                {error: 'username'} (Username not registered)
                {error: 'password'} (Incorrect password)

        /logout
            {token}
                {}

        /validate
            {token}
                {validity}

        /upload
            {token, moduleName, file}
                {success}

        /download
            {token, moduleName}
                {file}

        /rename
            {token, oldModuleName, newModuleName}
                {success}

        /remove
            {token, moduleName}
                {file}

        /user
            {token}
                {user}
